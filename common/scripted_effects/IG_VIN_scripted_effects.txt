VIN_modify_loyalty = {
	add_to_variable = { sphere_loyalty = temp_value }

	# remove old loyalty payment value
	subtract_from_variable = { VIN.total_loyalty_payments = loyalty_payment }

	# set expenses currently needed for loyalty
	set_variable = { loyalty_payment = max_loyalty_payment }
	multiply_variable = { loyalty_payment = sphere_loyalty }
	divide_variable = { loyalty_payment = 100 }
	round_variable = loyalty_payment

	clamp_variable = {
		var = loyalty_payment
		min = 0
		max = max_loyalty_payment
	}

	add_to_variable = { VIN.total_loyalty_payments = loyalty_payment }

	# scroll position
	set_variable = { slider_x_position = 72 }
	set_temp_variable = { temp_x = sphere_loyalty }
	multiply_temp_variable = { temp_x = 245 }
	divide_temp_variable = { temp_x = 100 }
	round_temp_variable = temp_x
	add_to_variable = { slider_x_position = temp_x }

	clamp_variable = {
		var = slider_x_position
		max = 317
	}

	# refresh gui
	add_to_variable = { VIN.show_sphere_subject_dirty = 1 }
}

# sets up the target for the sphere interactions screen
VIN_sphere_subject_display = {
	if = {
		limit = {
			check_variable = { temp = show_VIN_sphere_subject }
		}
		clear_variable = show_VIN_sphere_subject
		clear_array = sphere_subject_target_array
	}
	else = {
		set_variable = { show_VIN_sphere_subject = temp }
		clear_array = sphere_subject_target_array
		add_to_array = { sphere_subject_target_array = show_VIN_sphere_subject }
		add_to_variable = { show_sphere_subject_dirty = 1 }
	}
}

# Usage Example
# <SUBJECT_ID> = {
# 	set_temp_variable = { temp_loyalty_cost = 2 } # in thousands
# 	VIN_modify_loyalty_cost = yes
# }

# takes in two arguments (temp_loyalty_cost, temp_subject), modifying payment needed
VIN_modify_loyalty_cost = {
	custom_effect_tooltip = VIN_modify_loyalty_cost_tt

	add_to_variable = { max_loyalty_payment = temp_loyalty_cost }

	# make sure payment does not reach 0
	clamp_variable = {
		var = max_loyalty_payment
		min = 1
	}

	# refresh gui
	if = {
		limit = {
			check_variable = { VIN.show_sphere_subject_dirty > 0 }
		}
		add_to_variable = { VIN.show_sphere_subject_dirty = 1 }
	}
}

	}
}
